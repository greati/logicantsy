59  void sequents_from_determinants(const Determinant<std::set<int>>& det,
60  std::vector<NdSequent<std::set>>& sequents) {
61  auto response_set = det.get_last();
62  auto avoid_set = utils::set_difference(this->_all_values, response_set);
63  auto avoid_set_size = avoid_set.size();
64  auto all_values_size = _all_values.size();
65  // fully non-deterministic cell, no sequents is necessary
66  if (avoid_set_size == 0)
67  return;
68  // partiality
69  if (avoid_set_size == all_values_size) {
70  NdSequent<std::set> seq {_sequent_dim, {}};
71  place_variables(seq, det);
72  sequents.push_back(seq);
73  }
74  // other cases
75  else {
76  auto judgs_from_set = _judg_value_correspondence->judgements_from_set_values(avoid_set);
77  for (const auto& judgs : judgs_from_set) {
78  NdSequent<std::set> seq {_sequent_dim, {}};
79  place_in_sequent(seq, _compound, judgs);
