272  class EqualityFormulaVisitor : public FormulaVisitor<bool> {
273  private:
274 
275  std::shared_ptr<Formula> _left;
276 
277  public:
278  EqualityFormulaVisitor(decltype(_left) left) : _left {left} {};
279 
280  virtual bool visit_prop(Prop* prop) override {
281  if (_left->type() != Formula::FmlaType::PROP)
282  return false;
283  auto left_prop = std::dynamic_pointer_cast<Prop>(_left);
284  return left_prop->symbol() == prop->symbol();
285  }
286 
287  virtual bool visit_compound(Compound* compound) override {
288  if (_left->type() != Formula::FmlaType::COMPOUND)
289  return false;
290  auto left = std::dynamic_pointer_cast<Compound>(_left);
291  auto left_conn = left->connective();
292  if (*(compound->connective()) != *(left_conn))
293  return false;
294  auto left_comps = left->components();
295  auto right_comps = compound->components();
296  for (int i = 0; i < left_comps.size(); ++i) {
297  EqualityFormulaVisitor eqvis {left_comps[i]};
298  if (not right_comps[i]->accept(eqvis))
299  return false;
300  }
301  return true;
302  }
