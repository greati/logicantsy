553  static void construct_combination(combination& data, size_type m, IntType n)
554  {
555  IntType k = data.size();
556  m = binomial<size_type>(n, k) - m - 1;
557 
558  for (IntType i = 0; i < k; ++i)
559  {
560  IntType r = k - i;
561 
562  // i <= n-t-1 <= n-r implies that the range is this
563  big_integer_interval N(r, n - i);
564  auto t = N.partition_point([m, r](auto t) {
565  return binomial<size_type>(t, r) <= m;
566  }) -
567  1;
568 
569  data[i] = n - t - 1;
570  m -= binomial<size_type>(t, r);
571  }

616  bool augment(combination& comb, P pred, IntType start = 0)
617  {
