79  std::vector<std::vector<int>> process_tt_restriction(
80  std::vector<std::string>& rests,
81  const std::set<int>& values,
82  const std::map<std::string, int>& str_to_val) const {
83  if (rests.empty()) {
84  auto new_args = std::vector<std::vector<int>>();
85  new_args.push_back(std::vector<int>());
86  return new_args;
87  }
88  auto last_element = rests.back();
89  rests.pop_back();
90  auto restriction_recursive = process_tt_restriction(rests, values, str_to_val);
91  std::vector<std::vector<int>> r;
92  for (const auto& args_recursive : restriction_recursive) {
93  if (last_element == "_") {
94  for (const auto& v : values) {
95  auto new_args = std::vector<int>(args_recursive.begin(), args_recursive.end());
96  new_args.push_back(v);
97  r.push_back(new_args);
98  }
99  } else {
100  if (str_to_val.find(last_element) == str_to_val.end())
101  throw ParseException("value " + last_element + " not defined");
102  auto new_args = std::vector<int>(args_recursive.begin(), args_recursive.end());
103  new_args.push_back(str_to_val.find(last_element)->second);
104  r.push_back(new_args);
105  }
106  }
107  return r;
108  }
