684  static inline void construct_combination(combination& data, size_type m)
685  {
686  IntType k = data.size();
687 
688  // this is the biggest for which binomial is still well defined.
689  // Hopefully it's enough for most use cases.
690  size_type upper = 68;
691 
692  for (IntType r = k; r > 1; --r)
693  {
694  IntType t;
695 
696  big_integer_interval NR(r, upper);
697 
698  t = NR.partition_point(
699  [m, r](auto x) { return binomial<size_type>(x, r) <= m; }) -
700  1;

703  m -= binomial<size_type>(t, r);
704  }

753  template <class P>
754  bool augment(combination& comb, P pred, IntType start = 0)
755  {
756  if (comb.empty())
757  {
758  if (start < n_ - k_ + 1)
759  {
760  comb.push_back(start);
761  return true;
762  }
763  return false;

765 
766  auto last = comb.back();
767  auto guysleft = k_ - comb.size();
768 
769  start = std::max(static_cast<IntType>(last + 1), start);
770 
771  for (size_t i = start; i < n_ - guysleft + 1; ++i)
772  {
773  comb.push_back(i);

778  comb.pop_back();
