26  void handle(const std::string& yaml_path) {
27  YAMLCppParser parser;
28  try {
29  auto root = parser.load_from_file(yaml_path);
30  auto semantics_node = parser.hard_require(root, SEMANTICS_TITLE);
31  // parse values
32  auto values_node = parser.hard_require(semantics_node, SEMANTICS_VALUES_TITLE);
33  auto values = values_node.as<std::vector<std::string>>();
34  std::set<int> real_values;
35  auto nvalues = values.size();
36  for (int i {0}; i < nvalues; ++i) {
37  _str_to_val[values[i]] = i;
38  _val_to_str[i] = values[i];
39  real_values.insert(i);
40  }
41  // parse judgement names
42  auto judgements_node = parser.hard_require(semantics_node, SEMANTICS_JUDGMENTS_TITLE);
43  auto judgements_names = judgements_node.as<std::vector<std::string>>();
44  // semantics
45  std::shared_ptr<JudgementValueCorrespondence> jvc =
46  std::make_shared<FourBillatice>();
47  // semantics judgements
48  std::vector<CognitiveAttitude> judgements_in_semantics;
49  for (const auto& s : judgements_names)
50  judgements_in_semantics.push_back(jvc->get_judgement(s));
51  //parse tts and run app
52  auto tt_root = parser.hard_require(root, TRUTH_TABLE_TITLE);
53  for (const auto& tt_node : tt_root) {
54  // capture connective and variables
55  auto fmla_parser = parser.make_fmla_parser(tt_node.first);
56  auto conn_compound = tt_node.first.as<std::string>();
57  auto compound = std::dynamic_pointer_cast<Compound>(fmla_parser->parse(conn_compound));
58  auto conn_name = compound->connective()->symbol();
59  VariableCollector var_collector;
60  compound->accept(var_collector);
61  auto collected_vars = var_collector.get_collected_variables();
62  std::vector<std::shared_ptr<Prop>> vars;
63  for (const auto v : collected_vars)
64  vars.push_back(std::make_shared<Prop>(*v));
65  _connectives[conn_name] = compound->connective();
66  _props[conn_name] = vars;
67  // parse tt
68  NDTruthTable tt = parser.parse_nd_truth_table(tt_node.second, real_values,
69  compound->connective()->arity(), _str_to_val);
70  auto table_print = tt.print(_val_to_str).str();
71  spdlog::info("Input table for connective " + conn_name + " is\n" + table_print);
72  AppsFacade apps_facade;
73  auto sequents = apps_facade.axiomatize_truth_table(real_values,
74  *compound->connective(),
75  vars,
76  compound,
77  judgements_in_semantics,
78  jvc,
79  tt);
80  std::string out = "";
81  for (const auto& seq : sequents)
82  out += "- " + seq.to_string() + "\n";
83  spdlog::info("Here is an axiomatization for " + conn_name + ", with "
84  + std::to_string(sequents.size()) + " sequent(s):\n" + out);
85  }
86  spdlog::info("Done.");
87  } catch (ParseException& pe) {
88  spdlog::critical(pe.message());
89  } catch (YAML::ParserException& ye) {
90  spdlog::critical(ye.what());
91  }
92  }
