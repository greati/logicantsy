128 struct print_container_helper

140  {

142  stream << delimiters_type::values.prefix;
143  std::copy(_container->begin(), _container->end(), pretty_ostream_iterator<typename T::value_type, TChar, TCharTraits>(stream, delimiters_type::values.delimiter));
